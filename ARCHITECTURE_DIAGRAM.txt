  // Основная информация
  id: number,              // Уникальный ID товара
  name: string,            // Название товара (например, "iPhone 15 Pro Max")
  category: string,        // Категория (all|smartphones|laptops|components)
  price: string,           // Цена с форматированием (например, "450 000 ₸")
  rating: number,          // Рейтинг (0-5)
  specs: string,           // Краткие характеристики (одна строка)
  stock: string,           // Склады доступности (например, "Almaty, Astana")
  
  // Подробные характеристики для сравнения
  fullSpecs: [
    {
      category: string,    // Категория (Процессор, Память, Дисплей, и т.д.)
      label: string,       // Название параметра (Чип, ОЗУ, Размер экрана)
      value: string,       // Значение параметра (A17 Pro, 6GB, 6.7 дюймов)
      hasFeature: boolean  // Это особое преимущество? (true/false)
    },
    // ... остальные параметры
  ]
}
